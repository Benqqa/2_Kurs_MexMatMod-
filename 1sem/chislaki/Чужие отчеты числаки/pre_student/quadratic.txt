#include <stdio.h> //библиотека функций ввода/вывода
//#include <stdlib.h>
#include <math.h> //библиотека математических функциё( в данной программы используем функцию 'sqrt' для извлечения квадратного корня)
#include <locale.h> //библиотека для подлкючения поддержки русского языка
#pragma warning(disable:4996)

void f(int a, int b, int c) { //Функция, которая принимает в качестве параметров коэффициенты квадратного уравнения
  double d; //дискриминант
  double x1, x2; //корни уравнения
  d = (b * b) - 4 * a * c;//вычисление дискриминанта
  printf("Дискриминант равен %lf =>\n", d);
  if (d > 0) { //алгоритм дейсвий, если дискриминант больше нуля
    x1 = (-b + sqrt(d)) / (2 * a); //вычисление первого корня
    x2 = (-b - sqrt(d)) / (2 * a);//вычисление второго корня
    printf("Решение уровнения:\n");
    printf("\tx1 = %f\n", x1); //печать первого корня
    printf("\tx2 = %f\n", x2); //печать второго корня
  }
  if (d < 0) { //алгоритм дейсвий, если дискриминант меньше нуля
    printf("Уравнение не содержит вещественные корни!\n");
  }
  if (d == 0) {//алгоритм дейсвий, если дискриминант равен нулю
    x1 = (-b + sqrt(d)) / (2 * a);
    printf("Решение уровнения:\n");
    printf("\tx1 = x2 = %f\n", x1);
  }
}

int main(void) {
  int a = 0, b = 0, c = 0; //коэффициенты квадратного уравнения
  setlocale(LC_ALL, "Rus"); //подключение поддержки русского языка
  printf("Квадратное уравнение задается в виде: a*x^2 + b*x + c = 0\n");
  printf("Введите коэффициенты a,b,c\n");
  scanf("%i %i %i", &a, &b, &c); //считывание введенных коэффициентов с консоли
  f(a, b, c); //вызов "нашей" функции
  return 0; //завершение программы
}
